import turtle

# Set up the screen
screen = turtle.Screen()
screen.bgcolor("white")
screen.title("Heart Drawing")

# Set up the turtle pen
pen = turtle.Turtle()
pen.color("red")
pen.fillcolor("red")
pen.speed(3)

def draw_curve():
    """Draws the smooth curve part of the heart."""
    for _ in range(200):
        pen.right(1)
        pen.forward(1)

def draw_heart(name):
    """Draws the entire filled heart shape and writes the name."""
    pen.begin_fill()
    pen.left(140)
    pen.forward(113)
    draw_curve()
    pen.left(120)
    draw_curve()
    pen.forward(112)
    pen.end_fill()
    
    # Move the pen to write the name inside the heart
    pen.up()
    # Adjust y-coordinate based on where you want the text to sit
    pen.setpos(0, 90) 
    pen.color("white")
    pen.write(name, align="center", font=("Arial", 16, "bold"))
    pen.hideturtle()

# Change "Your Name" to whatever you prefer!
draw_heart("Pruthvi")

# Keep the window open until clicked
screen.exitonclick()
